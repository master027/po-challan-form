function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var data = JSON.parse(e.postData.contents); var challanLink = uploadFile(data.challan_file, data.challan_name, 'Challans'); var invoiceLink = uploadFile(data.invoice_file, data.invoice_name, 'Invoices'); var photoLinks = []; if(data.photos && data.photos.length > 0){ data.photos.forEach(function(p, i){ photoLinks.push(uploadFile(p.data, p.name, 'Photos')); }); } sheet.appendRow([ new Date(), data.po_no, data.vendor, data.material_name, data.material_other, data.material_brand, data.quantity, data.unit, data.qty_verified_by, data.qty_shortage, data.material_condition, data.visual_inspection, data.brand_verified, data.quality_result, challanLink, photoLinks.join('\n'), invoiceLink, data.engineer_name, data.po_completed, data.remarks ]); return ContentService.createTextOutput(JSON.stringify({status:"success"})) .setMimeType(ContentService.MimeType.JSON); } function uploadFile(base64Data, fileName, folderName) { if(!base64Data || base64Data === '') return 'No file'; try { var folder; var folders = DriveApp.getFoldersByName(folderName); if(folders.hasNext()){ folder = folders.next(); } else { folder = DriveApp.createFolder(folderName); } var blob = Utilities.newBlob( Utilities.base64Decode(base64Data), 'application/octet-stream', fileName ); var file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); return file.getUrl(); } catch(err) { return 'Upload failed: ' + err.message; } }
